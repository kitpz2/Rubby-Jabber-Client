  def wyslijrozmowa(text)
        polecenie=text.split
    if polecenie[0]=="!koniec" then
            @rozmowa=0
    else
                do_wyslania=String.new("")
                text.each{|txt|
                        do_wyslania+=txt
                        do_wyslania+=" "
                }
                @uzytkownik.wyslij_msg(@rozmowa_z,text,:rozmowa)
                zapisz_historie(@rozmowa_z.to_s,"#{@jid}:\n#{text}")
        end
  end
